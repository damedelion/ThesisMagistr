% Библиография: отступы и межстрочный интервал
\makeatletter
\renewenvironment{thebibliography}[1]
    {\section*{\refname}
        \list{\@biblabel{\@arabic\c@enumiv}}
           {\settowidth\labelwidth{\@biblabel{#1}}
            \leftmargin\labelsep
            \itemindent 16.7mm
            \@openbib@code
            \usecounter{enumiv}
            \let\p@enumiv\@empty
            \renewcommand\theenumiv{\@arabic\c@enumiv}
            % Убираем все вертикальные промежутки
            \setlength{\itemsep}{0pt}      % между пунктами
            \setlength{\parsep}{0pt}       % между абзацами внутри пункта
            \setlength{\topsep}{0pt}       % сверху/снизу списка
            \setlength{\partopsep}{0pt}    % при разрыве списка на страницах
        }
    }
\makeatother
